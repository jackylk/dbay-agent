# Wiki Lint 健康检查 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a Lint (health check) feature to wiki that detects 6 categories of quality issues, displays results in a side panel, and supports per-category and bulk fix via Curate.

**Architecture:** Backend adds `runLint` method to WikiService (rule checks + LLM analysis) and `fixLintIssues` (delegates to Curate with lint context). Frontend adds WikiLintPanel.vue as a right-side slide-out panel in the wiki tab, with issue badges on the wiki page list.

**Tech Stack:** Java 17, Spring Boot, DeepSeek V3.2 (Huawei MaaS), Vue 3 Composition API, TypeScript

```java
/**
 * Run lint checks on wiki pages. Returns issues grouped by category.
 * Phase 1: rule-based checks (language, orphan, broken_link) — no LLM needed.
 * Phase 2: LLM analysis (contradiction, stale, missing_link) — calls DeepSeek.
 */
public Map<String, Object> runLint(String tenantId, String kbId) {
    KnowledgeBaseEntity kb = kbAccessService.getKbWithAccess(kbId, tenantId);
    tenantId = kb.getTenantId();

    List<DocumentEntity> wikiDocs = documentRepository.findByTenantIdAndKbIdAndDocType(
            tenantId, kbId, DOC_TYPE_WIKI);

    // Load all page titles and content
    Map<String, String> pageContents = new LinkedHashMap<>();
    for (DocumentEntity doc : wikiDocs) {
        String fn = doc.getFilename();
        if ("index.md".equals(fn) || "log.md".equals(fn)) continue;
        String title = fn.replace(".md", "").replace("_", " ");
        String content = readWikiPage(tenantId, kbId, fn);
        if (content != null) pageContents.put(title, content);
    }

    List<Map<String, Object>> issues = new ArrayList<>();

    // 1. Language check: detect non-Chinese pages
    for (var entry : pageContents.entrySet()) {
        String title = entry.getKey();
        String content = entry.getValue();
        double chineseRatio = chineseCharRatio(content);
        if (chineseRatio < 0.3) {
            issues.add(Map.of(
                "category", "language", "severity", "error",
                "page", title, "description", "整页非中文，应翻译为中文",
                "related_pages", List.of()));
        } else if (chineseRatio < 0.7) {
            issues.add(Map.of(
                "category", "language", "severity", "warning",
                "page", title, "description", "部分内容为非中文",
                "related_pages", List.of()));
        }
    }

    // 2. Orphan check: pages with zero inbound links
    Map<String, Set<String>> inboundLinks = new HashMap<>();
    for (String title : pageContents.keySet()) {
        inboundLinks.put(title.toLowerCase(), new HashSet<>());
    }
    for (var entry : pageContents.entrySet()) {
        List<String> links = extractWikilinks(entry.getValue());
        for (String link : links) {
            Set<String> sources = inboundLinks.get(link.toLowerCase());
            if (sources != null) sources.add(entry.getKey());
        }
    }
    for (var entry : inboundLinks.entrySet()) {
        if (entry.getValue().isEmpty()) {
            String title = pageContents.keySet().stream()
                    .filter(t -> t.toLowerCase().equals(entry.getKey()))
                    .findFirst().orElse(entry.getKey());
            issues.add(Map.of(
                "category", "orphan", "severity", "warning",
                "page", title, "description", "无其他页面链接到此页",
                "related_pages", List.of()));
        }
    }

    // 3. Broken link check: wikilinks pointing to non-existent pages
    Set<String> existingTitles = pageContents.keySet().stream()
            .map(String::toLowerCase).collect(java.util.stream.Collectors.toSet());
    Map<String, List<String>> brokenLinks = new LinkedHashMap<>();
    for (var entry : pageContents.entrySet()) {
        for (String link : extractWikilinks(entry.getValue())) {
            if (!existingTitles.contains(link.toLowerCase())) {
                brokenLinks.computeIfAbsent(link, k -> new ArrayList<>()).add(entry.getKey());
            }
        }
    }
    for (var entry : brokenLinks.entrySet()) {
        issues.add(Map.of(
            "category", "broken_link", "severity", "error",
            "page", "", "description", "[[" + entry.getKey() + "]] 被 " + entry.getValue().size() + " 个页面引用但不存在",
            "related_pages", entry.getValue()));
    }

    // Build summary
    Map<String, Long> summary = issues.stream()
            .collect(java.util.stream.Collectors.groupingBy(
                    i -> (String) i.get("category"), java.util.stream.Collectors.counting()));
    summary.putIfAbsent("language", 0L);
    summary.putIfAbsent("orphan", 0L);
    summary.putIfAbsent("broken_link", 0L);
    summary.putIfAbsent("contradiction", 0L);
    summary.putIfAbsent("stale", 0L);
    summary.putIfAbsent("missing_link", 0L);
    summary.put("total", (long) issues.size());

    Map<String, Object> result = new LinkedHashMap<>();
    result.put("issues", issues);
    result.put("summary", summary);
    result.put("checked_at", java.time.Instant.now().toString());
    return result;
}

/** Calculate the ratio of Chinese characters in text (0.0 to 1.0). */
private double chineseCharRatio(String text) {
    if (text == null || text.isBlank()) return 1.0;
    // Strip markdown syntax for cleaner analysis
    String clean = text.replaceAll("```[\\s\\S]*?```", "")
            .replaceAll("`[^`]*`", "")
            .replaceAll("\\[\\[.*?]]", "")
            .replaceAll("[#*\\-|>\\[\\](){}]", "")
            .replaceAll("https?://\\S+", "")
            .trim();
    if (clean.isEmpty()) return 1.0;
    long total = clean.codePoints().filter(cp -> !Character.isWhitespace(cp)).count();
    if (total == 0) return 1.0;
    long chinese = clean.codePoints().filter(cp ->
            Character.UnicodeScript.of(cp) == Character.UnicodeScript.HAN).count();
    return (double) chinese / total;
}
```

```java
/**
 * LLM-based lint: detect contradictions, stale content, missing cross-references.
 */
private List<Map<String, Object>> runLlmLintAnalysis(String tenantId, String kbId,
                                                      Map<String, String> pageContents) {
    List<Map<String, Object>> issues = new ArrayList<>();
    if (pageContents.size() < 2) return issues; // Need at least 2 pages to compare

    String indexContent = readWikiPage(tenantId, kbId, "index.md");
    if (indexContent == null) indexContent = "";

    // Build page summaries (first 500 chars each, cap total at 30000)
    StringBuilder summaries = new StringBuilder();
    int totalChars = 0;
    for (var entry : pageContents.entrySet()) {
        if (totalChars >= 30_000) break;
        String excerpt = entry.getValue().substring(0, Math.min(entry.getValue().length(), 500));
        summaries.append("### ").append(entry.getKey()).append("\n").append(excerpt).append("\n\n");
        totalChars += excerpt.length() + entry.getKey().length() + 10;
    }

    String prompt = String.format("""
            You are a wiki quality auditor. Analyze the following wiki pages for quality issues.

            Current wiki index:
            ---
            %s
            ---

            Wiki page summaries:
            ---
            %s
            ---

            Check for:
            1. CONTRADICTIONS: Different pages describe the same concept differently or give conflicting facts.
            2. STALE: Claims that appear outdated because other pages contain newer or corrected information.
            3. MISSING_LINKS: Pages that discuss closely related topics but don't link to each other via [[wikilink]].

            Output ONLY valid JSON (no markdown fences):
            {
              "contradictions": [
                {"page": "Page A", "related_page": "Page B", "description": "具体矛盾描述（中文）"}
              ],
              "stale": [
                {"page": "Page", "description": "过时内容描述（中文）"}
              ],
              "missing_links": [
                {"page": "Page A", "related_page": "Page B", "description": "为什么应该互相链接（中文）"}
              ]
            }

            If no issues found for a category, use an empty array.
            Write all descriptions in Simplified Chinese.
            """, indexContent.substring(0, Math.min(indexContent.length(), 4000)), summaries);

    try {
        String baseUrl = getWikiBaseUrl();
        String apiKey = getWikiApiKey();
        String model = getWikiModel();

        Map<String, Object> requestBody = new LinkedHashMap<>();
        requestBody.put("model", model);
        requestBody.put("messages", List.of(
                Map.of("role", "user", "content", prompt)));
        requestBody.put("temperature", 0.1);
        requestBody.put("max_tokens", 4000);
        requestBody.put("response_format", Map.of("type", "json_object"));

        String body = new com.fasterxml.jackson.databind.ObjectMapper()
                .writeValueAsString(requestBody);

        java.net.http.HttpRequest request = java.net.http.HttpRequest.newBuilder()
                .uri(java.net.URI.create(baseUrl + "/chat/completions"))
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer " + apiKey)
                .POST(java.net.http.HttpRequest.BodyPublishers.ofString(body))
                .timeout(java.time.Duration.ofSeconds(60))
                .build();

        java.net.http.HttpResponse<String> response = java.net.http.HttpClient.newHttpClient()
                .send(request, java.net.http.HttpResponse.BodyHandlers.ofString());

        if (response.statusCode() != 200) {
            log.warn("Lint LLM call failed: {}", response.statusCode());
            return issues;
        }

        var om = new com.fasterxml.jackson.databind.ObjectMapper();
        var root = om.readTree(response.body());
        String content = root.path("choices").path(0).path("message").path("content").asText("");
        var parsed = om.readTree(content);

        for (var c : parsed.withArray("contradictions")) {
            issues.add(Map.of(
                "category", "contradiction", "severity", "error",
                "page", c.path("page").asText(""),
                "description", c.path("description").asText(""),
                "related_pages", List.of(c.path("related_page").asText(""))));
        }
        for (var s : parsed.withArray("stale")) {
            issues.add(Map.of(
                "category", "stale", "severity", "warning",
                "page", s.path("page").asText(""),
                "description", s.path("description").asText(""),
                "related_pages", List.of()));
        }
        for (var m : parsed.withArray("missing_links")) {
            issues.add(Map.of(
                "category", "missing_link", "severity", "info",
                "page", m.path("page").asText(""),
                "description", m.path("description").asText(""),
                "related_pages", List.of(m.path("related_page").asText(""))));
        }
    } catch (Exception e) {
        log.warn("Lint LLM analysis failed: {}", e.getMessage());
    }

    return issues;
}
```

```java
/**
 * Fix lint issues by running a targeted curate pass with lint context.
 * @param categories which issue categories to fix (empty = all)
 * @param issues the lint issues to fix (passed as context to curate prompt)
 */
public Map<String, Object> fixLintIssues(String tenantId, String kbId,
                                          List<String> categories, List<Map<String, Object>> issues) {
    KnowledgeBaseEntity kb = kbAccessService.getKbWithAccess(kbId, tenantId);
    tenantId = kb.getTenantId();

    // Filter issues by categories if specified
    List<Map<String, Object>> targetIssues = issues;
    if (categories != null && !categories.isEmpty()) {
        Set<String> catSet = new HashSet<>(categories);
        targetIssues = issues.stream()
                .filter(i -> catSet.contains(i.get("category")))
                .toList();
    }

    if (targetIssues.isEmpty()) {
        return Map.of("fixed", 0, "pages_updated", List.of(),
                "pages_created", List.of(), "pages_deleted", List.of());
    }

    // Build lint context for curate prompt
    StringBuilder lintContext = new StringBuilder("The following quality issues were found by lint. Fix them:\n\n");
    for (var issue : targetIssues) {
        lintContext.append("- [").append(issue.get("category")).append("] ");
        if (issue.get("page") != null && !issue.get("page").toString().isEmpty()) {
            lintContext.append("Page \"").append(issue.get("page")).append("\": ");
        }
        lintContext.append(issue.get("description")).append("\n");
    }

    // Run curate with lint context appended
    appendToLog(tenantId, kbId, "[Lint 修复] 修复 " + targetIssues.size() + " 个问题");

    // Load current wiki state (same as runCurate)
    String indexContent = readWikiPage(tenantId, kbId, "index.md");
    if (indexContent == null) indexContent = "# Wiki Index\n\nNo pages yet.\n";
    if (indexContent.length() > MAX_INDEX_CHARS) {
        indexContent = indexContent.substring(0, MAX_INDEX_CHARS);
    }

    List<DocumentEntity> wikiDocs = documentRepository.findByTenantIdAndKbIdAndDocType(
            tenantId, kbId, DOC_TYPE_WIKI);
    StringBuilder pagesContext = new StringBuilder();
    int totalChars = 0;
    for (DocumentEntity doc : wikiDocs) {
        if ("index.md".equals(doc.getFilename()) || "log.md".equals(doc.getFilename())) continue;
        if (totalChars >= 60_000) break;
        String content = readWikiPage(tenantId, kbId, doc.getFilename());
        if (content == null || content.isBlank()) continue;
        String title = doc.getFilename().replace(".md", "");
        String excerpt = content.substring(0, Math.min(content.length(), 1_500));
        pagesContext.append("### ").append(title).append("\n").append(excerpt).append("\n\n");
        totalChars += excerpt.length();
    }

    // Use curate prompt with lint additions
    String curatePrompt = String.format(DEFAULT_CURATE_PROMPT, indexContent, pagesContext);
    curatePrompt += "\n\nADDITIONAL CONTEXT — Lint issues to fix:\n" + lintContext;

    // Call LLM and apply changes (reuse curate response parsing logic)
    try {
        var response = callWikiAgent(tenantId, kbId, curatePrompt, true);
        int[] counts = applyCurateResponse(tenantId, kbId, response);
        return Map.of("fixed", targetIssues.size(),
                "pages_updated", counts[1], "pages_created", counts[0]);
    } catch (Exception e) {
        log.error("Lint fix failed: {}", e.getMessage());
        return Map.of("fixed", 0, "error", e.getMessage());
    }
}
```

Note: This method reuses the existing `callWikiAgent` and curate response parsing. The implementer should check how `runCurate` calls the LLM and parses the response, and either extract a shared method or call `runCurate` directly with modified prompt. The key is that `DEFAULT_CURATE_PROMPT` is a format string that takes index + pages content.

```vue
async function fixCategory(catKey: string) {
  fixing.value = catKey

    const catIssues = groupedIssues.value[catKey] || []

    emit('fixed')

    console.error('Fix failed:', e)

    fixing.value = null
  }
}

async function fixAll() {
  fixing.value = 'all'
  try {
    await fixWikiLint(props.kbId, [], props.issues)
    emit('fixed')
  } catch (e) {
    console.error('Fix all failed:', e)
  } finally {
    fixing.value = null
  }
}

function getCat(key: string) {
```
